# RM Systems — Astro Template

## Общая структура

```text
RM_Syst_Astro/
├── .env                         # переменные окружения проекта
├── Dockerfile                   # образ приложения
├── docker-compose.yml           # основной запуск сервисов
├── docker-compose.override.yml  # локальные настройки запуска
├── init_db.py                   # инициализация базы данных
├── api/                         # основной HTTP API
├── chat_api/                    # API чат-бота и рабочие часы
├── dao/                         # слой доступа к данным
├── deploy/                      # конфигурация развёртывания
├── INFO/                        # документация и описание проекта
├── scripts/                     # служебные скрипты
├── services/                    # интеграционные сервисы
│   ├── ai_gateway/              # шлюз AI-моделей
│   ├── bitrix_bridge/           # интеграция с Битрикс24
│   ├── leads_service/           # обработка заявок и лидов
│   └── rag_service/             # поиск по базе знаний RAG
├── shared/                      # общие модули приложения
│   ├── ai/                      # общая логика AI
│   ├── broadcast/               # рассылки и уведомления
│   ├── db/                      # модели и репозитории БД
│   ├── domain/                  # общие сущности и правила
│   ├── leads/                   # общая логика лидов
│   ├── prompts/                 # AI-промпты
│   ├── schedule/                # расписание и рабочие часы
│   ├── config.py                # общие настройки
│   ├── rag_profiles.py          # профили RAG
│   └── ttl_cache.py             # кэш с ограниченным сроком жизни
└── frontend/                    # публичный шаблон сайта на Astro
    ├── .env                     # SEO и публичные настройки сайта
    ├── .gitignore               # правила Git для frontend
    ├── public/                  # статические файлы сайта
    │   ├── favicons/            # favicon и иконки PWA
    │   └── fonts/               # шрифты интерфейса
    ├── src/
    │   ├── admin/               # только заглушка будущей админки
    │   ├── ai-widget/           # AI-виджет сайта
    │   ├── components/          # общие компоненты интерфейса
    │   ├── data/                # данные, регионы и SEO Schema.org
    │   ├── en/                  # англоязычная версия
    │   ├── layouts/             # базовые шаблоны страниц
    │   ├── pages/               # страницы и маршруты сайта
    │   ├── styles/              # глобальные и общие стили
    │   ├── utils/               # вспомогательные функции
    │   ├── env.d.ts             # типы окружения Astro
    │   └── middleware.ts        # middleware сайта
    ├── astro.config.mjs         # конфигурация Astro
    ├── eslint.config.mjs        # настройки ESLint
    ├── package.json             # зависимости и команды frontend
    ├── tsconfig.json            # настройки TypeScript
    └── tools/                   # вспомогательные инструменты проекта
```
